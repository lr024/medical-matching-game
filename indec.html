    <h1>Pathogen & Treatment Matchmaker</h1>

    <p>Select a disease on the left and match it with its correct treatment on the right!</p>

    <div id="message">Make your first match!</div>


    <div class="game-board">

        <div class="column" id="diseases-col"></div>

        <div class="column" id="cures-col"></div>

    </div>


    <script>

        const pairs = [

            { id: 1, disease: "Hepatitis C", cure: "Sofosbuvir" },

            { id: 2, disease: "HIV/AIDS", cure: "Antiretroviral Therapy (ART)" },

            { id: 3, disease: "Influenza (The Flu)", cure: "Oseltamivir (Tamiflu)" },

            { id: 4, disease: "Herpes Simplex (HSV)", cure: "Acyclovir" },

            { id: 5, disease: "COVID-19", cure: "Paxlovid" },

            { id: 6, disease: "Ebola Virus", cure: "Monoclonal Antibodies (Inmazeb)" },

            { id: 7, disease: "Rabies", cure: "Post-Exposure Prophylaxis (PEP)" },

            { id: 8, disease: "Chickenpox / Shingles", cure: "Valacyclovir" },

            { id: 9, disease: "Hepatitis B", cure: "Tenofovir" },

            { id: 10, disease: "Smallpox", cure: "Tecovirimat (TPOXX)" }

        ];


        let selectedDisease = null;

        let selectedCure = null;

        let matchedCount = 0;


        // Shuffle function

        function shuffle(array) {

            return array.sort(() => Math.random() - 0.5);

        }


        function initGame() {

            const diseasesCol = document.getElementById('diseases-col');

            const curesCol = document.getElementById('cures-col');

            const messageEl = document.getElementById('message');


            // Create shuffled arrays

            const diseases = shuffle([...pairs].map(p => ({ id: p.id, text: p.disease, type: 'disease' })));

            const cures = shuffle([...pairs].map(p => ({ id: p.id, text: p.cure, type: 'cure' })));


            // Render cards

            diseases.forEach(item => diseasesCol.appendChild(createCard(item)));

            cures.forEach(item => curesCol.appendChild(createCard(item)));


            function createCard(item) {

                const div = document.createElement('div');

                div.className = 'card';

                div.textContent = item.text;

                div.dataset.id = item.id;

                div.dataset.type = item.type;

                

                div.onclick = () => handleSelection(div, item);

                return div;

            }


            function handleSelection(element, item) {

                if (element.classList.contains('matched')) return;


                // Deselect if clicking the same item

                if (element.classList.contains('selected')) {

                    element.classList.remove('selected');

                    if (item.type === 'disease') selectedDisease = null;

                    else selectedCure = null;

                    return;

                }


                // Handle selection

                if (item.type === 'disease') {

                    if (selectedDisease) selectedDisease.element.classList.remove('selected');

                    selectedDisease = { element, id: item.id };

                } else {

                    if (selectedCure) selectedCure.element.classList.remove('selected');

                    selectedCure = { element, id: item.id };

                }

                

                element.classList.add('selected');


                // Check for match if both are selected

                if (selectedDisease && selectedCure) {

                    checkMatch();

                }

            }


            function checkMatch() {

                const dCard = selectedDisease.element;

                const cCard = selectedCure.element;


                if (selectedDisease.id === selectedCure.id) {

                    // Correct Match

                    dCard.className = 'card matched';

                    cCard.className = 'card matched';

                    messageEl.textContent = "Correct! Great job.";

                    messageEl.style.color = "#27ae60";

                    matchedCount++;


                    if (matchedCount === pairs.length) {

                        messageEl.textContent = "🎉 Congratulations! You matched them all! 🎉";

                    }

                } else {

                    // Wrong Match

                    dCard.classList.remove('selected');

                    cCard.classList.remove('selected');

                    dCard.classList.add('wrong');

                    cCard.classList.add('wrong');

                    messageEl.textContent = "Incorrect match. Try again!";

                    messageEl.style.color = "#e74c3c";


                    setTimeout(() => {

                        dCard.classList.remove('wrong');

                        cCard.classList.remove('wrong');

                    }, 500);

                }


                selectedDisease = null;

                selectedCure = null;

            }

        }


        // Start the game

        initGame();

    </script>
